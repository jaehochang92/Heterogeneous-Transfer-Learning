\begin{algorithm}
\caption{Cross-Validated High-Dimensional Sieve Estimation}
\label{alg:cv_sieve}
\begin{algorithmic}[1]
\REQUIRE $X \in \mathbb{R}^{n \times p}$, $y \in \mathbb{R}^n$, $\mathrm{basis\_numbers} \subset \mathbb{N}$ (candidate $J$), $n_{\mathrm{folds}} \in \mathbb{N}$, $\mathrm{type}$ (basis family)
\ENSURE $\lambda^* \in \mathbb{R}$, $J^* \in \mathbb{N}$, results $=\{\overline{\mathrm{MSE}}_{J,\ell}\}$
\STATE $n \gets |X|_{\text{rows}},\; p \gets |X|_{\text{cols}}$
\STATE Partition indices: $\{I_1,\dots,I_{n_{\mathrm{folds}}}\}$
\STATE $J_{\max} \gets \max(\mathrm{basis\_numbers})$
\STATE Precompute global penalty grids: for each $J\in\mathrm{basis\_numbers}$ compute
\[\lambda^{(J)} = \text{grid from }\text{sieve\_solver}(\text{sieve\_preprocess}(X,J_{\max}))\]
\FOR{$k=1$ to $n_{\mathrm{folds}}$}
    \STATE $I_{\text{valid}} \gets I_k,\; I_{\text{train}} \gets [n]\setminus I_k$
    \STATE $X_{\text{train}}=X[I_{\text{train}},:],\; y_{\text{train}}=y[I_{\text{train}}]$
    \STATE $X_{\text{valid}}=X[I_{\text{valid}},:],\; y_{\text{valid}}=y[I_{\text{valid}}]$
    \STATE $\Phi^{\text{full}}_{\text{train}}=\text{sieve\_preprocess}(X_{\text{train}},J_{\max})\in\mathbb{R}^{n_{\text{train}}\times J_{\max}}$
    \STATE $\Phi^{\text{full}}_{\text{valid}}=\text{sieve\_preprocess}(X_{\text{valid}},J_{\max})\in\mathbb{R}^{n_{\text{valid}}\times J_{\max}}$
    \FOR{each $J\in\mathrm{basis\_numbers}$}
        \STATE $\Phi_{\text{train},J}=\Phi^{\text{full}}_{\text{train}}[:,1:J]\in\mathbb{R}^{n_{\text{train}}\times J}$
        \STATE $\Phi_{\text{valid},J}=\Phi^{\text{full}}_{\text{valid}}[:,1:J]\in\mathbb{R}^{n_{\text{valid}}\times J}$
        \FOR{each index $\ell$ in $\{1,\dots,|\lambda^{(J)}|\}$}
            \STATE Fit $\hat{\beta}_{k,J,\ell} = \arg\min_{\beta}\; \frac{1}{n_{\text{train}}}\|y_{\text{train}}-\Phi_{\text{train},J}\beta\|_2^2 + \lambda^{(J)}_{\ell}\|\beta\|_1$
            \STATE $\mathrm{MSE}_{k,J,\ell} = \frac{1}{n_{\text{valid}}}\|y_{\text{valid}}-\Phi_{\text{valid},J}\hat{\beta}_{k,J,\ell}\|_2^2$
            \STATE Record $\mathrm{MSE}_{k,J,\ell}$
        \ENDFOR
    \ENDFOR
\ENDFOR
\STATE Aggregate: $\overline{\mathrm{MSE}}_{J,\ell} = \frac{1}{n_{\mathrm{folds}}}\sum_{k=1}^{n_{\mathrm{folds}}}\mathrm{MSE}_{k,J,\ell}$
\STATE $(J^*,\ell^*) = \arg\min_{J,\ell} \overline{\mathrm{MSE}}_{J,\ell}$
\STATE $\lambda^* = \lambda^{(J^*)}_{\ell^*}$
\RETURN $J^*,\lambda^*, \{\mathrm{MSE}_{k,J,\ell}\}, \{\overline{\mathrm{MSE}}_{J,\ell}\}$
\end{algorithmic}
\end{algorithm}
